gppt

gppt: get-pixivpy-token

  • Get your Pixiv token (for running upbit/pixivpy)
  • Refine pixiv_auth.py + its fork

Install

pip install gppt

Enable Proxy

Set ALL_PROXY or HTTPS_PROXY to your environment variables.

Run

Example

from gppt import GetPixivToken
from pixivpy3 import AppPixivAPI

def get_refresh_token() -> str:
    with open("token.txt", "w+") as f:
        if refresh_token := f.read().strip():
            return refresh_token

        g = GetPixivToken(headless=True)
        refresh_token = g.login(username="...", password="...")["refresh_token"]
        f.write(refresh_token)
        return refresh_token

aapi = AppPixivAPI()
aapi.auth(refresh_token=get_refresh_token())
...

 34    def auth(self) -> tuple[AppPixivAPI, LoginInfo]:
 35        """Authenticate the user with Pixiv.
 36
 37        This method attempts to log in to Pixiv using the provided credentials.
 38        It first checks for existing credentials in a JSON file. If the credentials
 39        are not found or if the login fails, it prompts the user for their ID and
 40        password. The method will retry the login process up to three times before
 41        raising a PixivLoginFailed exception.
 42
 43        Returns:
 44            tuple[AppPixivAPI, LoginInfo]: A tuple containing the authenticated
 45                AppPixivAPI instance and the login information.
 46
 47        Raises:
 48            PixivLoginFailed: If the login process fails after three attempts.
 49        """
 50        cnt = 0
 51        while cnt < 3:  # noqa: PLR2004
 52            try:
 53                aapi, login_info = self.__auth(cnt)
 54            except (ValueError, UnboundLocalError):
 55                print("\x1b[?25h[!]: Failed to login. Check your ID or PW.")
 56            else:
 57                return aapi, login_info
 58            cnt += 1
 59
 60        print("[!]: The number of login attempts has been exceeded.")
 61        raise PixivLoginFailedError
 62
 63    def __auth(self, cnt: int) -> tuple[AppPixivAPI, LoginInfo]:
 64        aapi: AppPixivAPI = AppPixivAPI()
 65        login_cred: LoginCred | None = self.read_client_cred()
 66
 67        if login_cred is not None and cnt == 0:
 68            ref = self.get_refresh_token(login_cred["pixiv_id"], login_cred["password"])
 69            print("\x1b[?25l[+]: Login...")
 70            login_info = aapi.auth(refresh_token=ref)
 71        elif login_cred is None or cnt > 0:
 72            print("[+]: ID is mail address, username, or account name.")
 73            stdin_login = (
 74                pwinput.pwinput(prompt="[?]: ID: ", mask=" "),
 75                pwinput.pwinput(prompt="[?]: PW: ", mask=" "),
 76            )
 77            print("\x1b[?25l[+]: Login...", end="\r")
 78            ref = self.get_refresh_token(stdin_login[0], stdin_login[1])
 79            login_info = aapi.auth(refresh_token=ref)
 80            print("\033[K[+]: Login...OK!\x1b[?25h")
 81        else:
 82            raise PixivLoginFailedError
 83
 84        return (aapi, login_info)
